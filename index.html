l.outerHtml = function(oh){
	if(typeof oh !== "undefined"){
		if(l.ia){
			for(var i = 0; i < this.length; i++){
				l[i].outerHTML = oh;
			}
		} else
			l.outerHTML = oh;
	} else{
		if(l.ia){
			var oha = [];
			for(var i = 0; i < this.length; i++){
				oha[i] = l[i].outerHTML;
			}
			return oha;
		} else {
			return l.outerHTML;
		}
	}
};
